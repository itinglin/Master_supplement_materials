def roc_calculate(y_predprob, y_test, threshold_num, ypredproblist):
    tprlist = []
    fprlist = []
    for k in range(threshold_num):
        tp = 0
        fp = 0
        tn = 0
        fn = 0
        for i in range(threshold_num): 
            if(y_predprob[i] >= ypredproblist[k]):
                if(y_test[i] == 1):
                    tp += 1
                else:
                    fp += 1
            elif(y_predprob[i] < ypredproblist[k]):
                if(y_test[i] == 0):
                    tn += 1
                else:
                    fn += 1
        tpr = tp/(tp+fn)
        fpr = fp/(fp+tn)
        tprlist.append(tpr)
        fprlist.append(fpr)
    return(tprlist, fprlist)

import math


class mylogistic_l2():
    def __init__(self, reg_vec, max_iter = 100, tol = 0.00001, add_intercept = True):
        """reg_vec: the regularization coefficient vector
           max_iter: maximum number of iteration to run for the Newton method
           tol: tolerance for the objective function
           add_intercept: whether to add intercept (a column of ones) at last column of the feature matrix"""
        self.reg_vec = reg_vec
        self.max_iter = 1000
        self.tol = tol
        self.add_intercept = add_intercept
            
    def fit(self, x, y, verbal = False):
        m, n = x.shape
        add = [1]*m
        add2 = np.array(add)
        # 依照題目限制更改lambda vector
        if self.add_intercept == True:
            x = np.column_stack((x, add2.T))
            
        if self.add_intercept == False:
            self.reg_vec = np.delete(self.reg_vec,-1,axis = 1)  #
        self.verbal = verbal
        self.m, self.n = x.shape #共有m列、n行
        self.w = np.zeros(self.n)
        self.m1, self.n2 = self.reg_vec.shape #103*103
        
        self.x = x
        self.y = y
        xx = self.x.T, self.y
        
        initial_w = np.dot(np.linalg.inv(np.dot(self.x.T, self.x)+np.dot(np.trace(self.reg_vec)/self.n2, np.eye(self.n))), np.dot(self.x.T, self.y))
        

        for i in range(self.max_iter):
            error_0 = 0
            if i == 0:
                w_old = initial_w
                for j in range(self.m):
                    sigmoid = 1 / (1 + np.exp(np.dot(-w_old.T, self.x[j])))
                    error_0 += self.y[j]*math.log(sigmoid) + (1-self.y[j])*math.log(1-sigmoid)
                error_old = np.dot(np.dot(w_old.T, self.reg_vec), w_old)/2 -error_0
                
            else:
                w_old = w_new
            Sigmoid = []
            Sigmoid_neg = []
            r = []
            
            for j in range(self.m):
                sigmoid = 1 / (1 + np.exp(np.dot(-w_old.T, self.x[j])))
                Sigmoid.append(sigmoid)
                sigmoid_neg = 1-sigmoid
                Sigmoid_neg.append(sigmoid_neg)
                error_0 += self.y[j]*math.log(sigmoid) + (1-self.y[j])*math.log(1-sigmoid)
            sigmoid_arr = np.array(Sigmoid)
            sigmoid_neg_arr = np.array(Sigmoid_neg)
            differ = sigmoid_arr - self.y
            difference = np.sum(differ)
            gradient = np.dot(self.reg_vec, w_old)+ np.dot(self.x.T, differ) 
            R = np.zeros(self.x.shape[0])
            for j in range(self.x.shape[0]):
                R[j] = sigmoid_arr[j] * sigmoid_neg_arr[j]
            R = np.diag(R)
            
            hessian = np.dot(np.dot(self.x.T, R), self.x) + self.reg_vec
            w_new = w_old - np.dot(np.linalg.inv(hessian), gradient)
            # 計算error
            error_1 = 0
            for j in range(self.m):
                sigmoid = 1 / (1 + np.exp(np.dot(-w_new.T, self.x[j])))
                error_1 += self.y[j]*math.log(sigmoid) + (1-self.y[j])*math.log(1-sigmoid)
            error_new = np.dot(np.dot(w_new.T, self.reg_vec), w_new)/2 -error_1
            
            if abs(error_new-error_old) < self.tol:
                break
            error_old = error_new
        self.optimize_w = w_new
    def predict(self, x_test):
        """doing prediction"""
        m, n = x_test.shape
        add = [1]*m
        add2 = np.array(add)
        ypredlist = []
        if self.add_intercept == True:
            x_test = np.column_stack((x_test, add2.T))
        for i in range(m):
            sigmoidpred = 1 / (1 + np.exp(np.dot(-self.optimize_w.T, x_test[i])))
            if sigmoidpred > 0.5:
                sigmoidpred = 1
            else:
                sigmoidpred = 0

            ypredlist.append(sigmoidpred)
        ypred = np.array(ypredlist)
        # print(self.optimize_w)
        return ypred
    def tuning(self, x_subtrain, y_subtrain, x_tuning, y_tuning):
        self.fit(x_subtrain, y_subtrain)
        ypredtune = self.predict(x_tuning)
        correct = 0 
        for i in range(y_tuning.shape[0]):
            if ypredtune[i] == y_tuning[i]:
                correct += 1
        accuracy = correct/y_tuning.shape[0]
        print(f'accuracy record: {accuracy}')
        return accuracy
