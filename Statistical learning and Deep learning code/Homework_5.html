# preprocessing 
# 由於圖片的解析度較高，模型訓練前須將解析度調整(Resize)成較短邊為256像素的照片，然後隨機取大小為224x224的影像。接著隨機水平翻轉(Horizontal Flip)、隨機旋轉-20度到20度，並依照Pretrained ResNet的要求調整RGB的均數與標準差。
# 測試資料(Valid and Test)亦須先將解析度調整(Resize)成較短邊為256像素的照片，然後取圖片中心224x224的影像。

preprocess_train = transforms.Compose([transforms.Resize(256),
    transforms.RandomCrop(224),
    transforms.RandomHorizontalFlip(),
    transforms.RandomRotation(20),
    transforms.ToTensor(),
    transforms.Normalize(mean=[0.485, 0.456, 0.406], std=[0.229, 0.224, 0.225]),      ])

preprocess_test = transforms.Compose([transforms.Resize(256),
    transforms.CenterCrop(224),
    transforms.ToTensor(),
    transforms.Normalize(mean=[0.485, 0.456, 0.406], std=[0.229, 0.224, 0.225])])

train_dataset = datasets.ImageFolder(root='/content/drive/MyDrive/photos/train', transform=preprocess_train)

import torch.nn as nn

class myresnet(): 
  def __init__(self):
    self.loss_func = nn.CrossEntropyLoss()
    self.valid_loss = []
    self.valid_accuracy = [] 
  def evaluate(self, model, data_loader):
    total_loss = 0
    total_correct = 0
    total_count = 0
    model.eval()
    with torch.no_grad():
      for inputs, labels in data_loader:
        outputs = model(inputs)
        total_loss += self.loss_func(outputs, labels)*labels.shape[0]
        total_count += len(outputs)
        x, preds = torch.max(outputs, dim=1)
        total_correct += torch.sum(preds == labels).item()
    model.train()
    loss_ratio = total_loss/total_count
    correct_ratio = total_correct/total_count
    return loss_ratio, correct_ratio
  def fit(self, model, train_loader, valid_loader, optimizer, epoch):
    best_model = model
    # 計算初始殘差
    valid_loss2, valid_accuracy2 = self.evaluate(model, valid_loader)
    self.valid_loss.append(valid_loss2)
    self.valid_accuracy.append(valid_accuracy2)

    top_epoch = 0
    top_loss = 10
    for i in range(epoch):
      # train the model
      for batch in train_loader:
        inputs, labels = batch
        optimizer.zero_grad()
        output = model(inputs)
        loss = self.loss_func(output, labels)
        loss.backward()
        optimizer.step()
      
      valid_loss2, valid_accuracy2 = self.evaluate(model, valid_loader)
      self.valid_loss.append(valid_loss2)
      self.valid_accuracy.append(valid_accuracy2)
      print(f'epoch{i}')
      print(valid_loss2, top_loss)
      if valid_loss2 < top_loss:
        top_epoch = i
        top_loss = valid_loss2
        best_model = model #
      if i - top_epoch >= 20:
        return best_model, self.valid_accuracy, self.valid_loss
    return best_model, self.valid_accuracy, self.valid_loss

  def test(self, model, data_loader):
    correct_sum = 0
    total_count = 0
    model.eval()
    y_pred = []
    y_expected = []
    with torch.no_grad():
      for inputs, labels in data_loader:
        outputs = model(inputs)
        total_count += len(outputs)
        x, preds = torch.max(outputs, dim=1)

        y_pred.append(preds.item())
        y_expected.append(labels.item())
        correct_sum += torch.sum(preds == labels).item()
    accuracy = correct_sum/total_count      
    print(f'Test Accuracy: {100*accuracy:.2f}%')

    # confusion matrix
    confusion_matrix = metrics.confusion_matrix(y_true=y_expected, y_pred = y_pred)
    cm_display = metrics.ConfusionMatrixDisplay(confusion_matrix = confusion_matrix, display_labels = ['blazer', 'cardigan', 'coat', 'jacket'])
    cm_display.plot(cmap=plt.cm.Oranges)
    plt.title('confusion matrix')
    plt.show()
    return confusion_matrix
